# count
> 변수에 대한 선언 <br>
> 변수를 선언할 경우 자료형의 크기만큼 메모리 공간을 확보하여 그 위치에 변수의 이름을 지정하게 됩니다.

+ 변수를 만들 때 지켜줄 이름 규칙 <br>
	1. 대소문자를 구분합니다 ex) a 와 A는 서로 다름 <br> 
	2. 변수의 이름은 숫자가 맨 앞에 나롱 수 없습니다. ex) int 1a; <--안됨 , int a1; <--됨 <br>
	3. 변수의 이름에 언더바 이외의 특수문자 사용이 불가능 합니다. (예외) _, $는 c언어에서 사용 가능 <br>
	4. 변수의 이름으로 키워드(keyword)를 사용할 수 없습니다.

```c 
#include <stdio.h>

int main()
{
	  int count; 
 //(자료형) (변수명);

	count = 5;
	//변수에 대한 초기화
	//변수명 = 타입에 맞는 값;
	
	printf("현제의 카운트는 %d입니다.\n", count);

	
	//변수는 선언과 초기화를 동시에 진행할 수 있습니다. 
	float pi = 3.14f;

	//데이터의 형태가 같다면, 여러 개를 한번에 선언할 수 있습니다. 
	int a, b, c;


	//선언과 초기화를 동시에 여러 개를 진행할 수 있습니다. 
	int number3 = 50, number4 = 40;

	//c언어에서 10진수 이외의 8진수나 16진수를 지정할 수 있습니다. 
	int octo = 015; //숫자 앞에 0이 붙으면 8진수
	int hexa = 0x6; //숫자 앞에 0x이 붙으면 16진수

	printf("8진수 octo = %o\n", octo);
	printf("16진수 hexa = %x\n", hexa);
	printf("8진수 표기법 출력 hexa = %#o\n", octo);
	printf("16진수 표기법 출력 hexa = %#x\n", hexa);
	printf("8진수 표기법 출력(8자리) octo : %0.8o\n", octo);
	printf("16진수 표기법 출력(8자리) hexa : %0.8x\n", hexa);

	return 0;
}
```

# 입력
> 입력(input) <br>
> 사용자가 직접 값을 입력하고 그 값을 특정 위치에 전달합니다.일반적으로는 변수를 만들고, 그 변수에 입력한 값을 적용하는 방식으로 사용합니다.

+ getchar() <br>
	> 키보드로부터 하나의 문자를 읽어내는 함수 <br>
	> 이 기능은 버퍼를 비우는 용도로도 사용됩니다. <br>
	> 버퍼(buffer) : c언어에서 입력을 진행할 경우 바로 값을 넘기는게 아니라 임시로 데이터를 저장하는 별도의 공간에 값을 넣고 전달하는 방식 <br>
	> 문자를 다 받기 때문에 enter같은 기능도 값으로 남게 됩니다. <br>

+ putchar(문자형 데이터) <br>
	> 전달받은 문자 하나를 화면에 출력하는 기능

+ gets(), puts() <br>
 > 주로 문자열에 대한 입출력을 진행할 때 편하게 사용가능 <br>
	  문자열이란 char 형태의 데이터를 묶음으로 표현한 것을 의미합니다. <br>
 > 문자열 만드는 방법 <br>
 > char 변수명[문자의 개수];

+ 주의사항 <br>
 > c언어에서 한글을 입력하는 경우 한글 하나당 2칸을 소모함 <br>
 >	현재 word는 10개의 단어를 사용가능하고 한글은 5개의 단어까지 입력 가능합니다. <br>
	  단, 입력을 진행할 경우 문자의 개수보다 1개정도 적게 작성합니다. ex) 10칸 사용할 경우 한글은 1칸 빼고 2로 나눠서 4칸밖에 사용못함 <br>
	  따라서 영단어는 9글자까지, 한글은 4글자까지 가능

# 연산자
> 프로그램에서 연산을 진행할 때 사용하는 기호
> 연산을 진행할 때 필요한 식이나 값을 항이라고 부름
> 예시) 2 + 5 라고 할 때 연산자는 + 이고, 위 식에서 항은 2, 5 두개임
> 항이 1개일땐 단항, 2개일땐 이항, 3개일땐 삼항이고 그 이상은 없음

+ 산술연산자
  |종류|역할|
  |-----|-----|
  |+|더하기|
  |-|빼기|
  |*|곱하기|
  |/|나누기|
  |%|나머지|

```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() 
{
	int a, b, c;
	a = 5;
	b = 3;

	c = a + b;
	printf("%d + %d = %d\n", a, b, c);
	// 5 + 3 = 8

	c = a - b;
	printf("%d - %d = %d\n", a, b, c);
	// 5 - 3 = 2

	c = a * b;
	printf("%d * %d = %d\n", a, b, c);
	// 5 * 3 = 15

	c = a / b;
	printf("%d / %d = %d\n", a, b, c);
	// 5 / 3 = 1

	c = a % b;
	printf("%d %% %d = %d\n", a, b, c);
	// 5 % 3 = 2

	return 0;
}
```

+ 증강연산자

> 변수의 값을 1 증가시킬 때 사용하는 연산자(단항) <br>
> 작성 위치에 따라 판정이 다르게 처리됩니다 <br>
> b = a++;일 경우를 후위형 증강 연산이라고 부르며, 이 경우에는 연산을 먼저 진행한 뒤, a의 값을 1 증가시킵니다. <br>
> b = ++a; 일 경우를 전위형 증강 연산이라고 부르며, 이 경우에는 a에 대한 1 증가를 먼저 진행하고, 연산을 처리합니다. <br>

```c
include <stdio.h>
int main()
{
	a = 10;
	b = a++;
	printf("a = %d, b = %d\n", a, b);
	//대입 연산에 의해 b는 a(10)값을 대입하게 되고, 그 후에 a가 1 증가합니다.
	// a = 11 , b = 10

	b = ++a;
	printf("a = %d, b = %d\n", a, b);
	//이 경우에는 a에 대한 1 증가를 먼저 진행하고, 연산을 처리합니다.
	// a = 12 , b = 12


return 0;
}

```



+ 관계연산자
 > 연산자를 기준으로 왼쪽과 오른쪽 값의 크기를 비교해서 참 또는 거짓을 확인합니다. <br>
 > 참일 경우 1을 거짓일 경우 0으로 처리됩니다.<br>
 > 관계연산자는 아래의 if문 같이 조건에 따라 코드를 작성하는 상황에 사용됩니다. <br>

```c
include <stdio.h>
int main()
{
	a = 10;
	b = 7;

	// if (조건식)
	// {
	// 조건식이 만족할 때, 실행할 명령문;
	// }
	
	if (a > b)
	{
		printf("a는 b보다 큰 값입니다.\n");
	}

	if ( a>= b) // a가 b보다 더 크거나 같은 경우를 확인합니다.
	{
		printf("a는 b보다 크거나 같습니다.\n");
	}

	if (a == b) // a와 b의 크기가 동일한지를 확인합니다.
	{
		printf("a와 b의 크기가 같습니다.\n");
	}

	if (a != b) // a와 b가 서로 다른 경우를 확인합니다
	{
		printf("a는 b와 다릅니다.\n");
	}
	
	if (a < b)
	{
		printf("a는 b보다 작습니다.\n");
	}
	
	if (a <= b) // a가 b보다 작거나 같은 경우를 확인합니다.
	{
		printf("a는 b보다 작거나 같습니다.\n");
	}

	

	return 0;
}

```

# 2진수 계산법
> 1. 계산할 값을 2의 제곱수로 분배합니다. <br>
> 2. 이때 2의 0제곱(1)도 범위에 포함됩니다. <br>
> 3. 가장 큰 값부터 잘라냅니다. <br>

+ 제곱 수
> 1 2 4 8 16 32 64 128 <br>
> 11 = 8 + 2 + 1 --> 1011 <br>
> 14 = 8 + 4 + 2 --> 1110 <br>

 # 기타 연산자

```c
#include <stdio.h>

int main()
{
	int a = 11;
	int b = 14;
	int c;


	printf("a = %d, b = %d\n", a, b); // 11 14
	c = a & b; //a AND b --> 교집합
	printf("a & b = %d\n", c); // 10
	//11 & 14는 11과 14가 가지고 있는 비트 중에서 겹치는 부분만 1로 처리, 나머지는 다 0으로
	//1011
	//1110
	//1010 --> 8 + 2 = 10



	c = a | b; //a OR b --> 합집합
	// a와 b의 값을 전부 1로 처리합니다, 이때 겹치는 값은 겹쳐집니다
	printf("a | b = %d\n", c); //15
	// 11  1011
	// 14  1110
	// c = 1111
	

	c = a ^ b; //a XOR b
	//XOR(exclusive OP\R : 배타적 논리합, 대칭 차집합
	//a와 b를 비교했을 때 비트가 다른 지점을 1, 같은 지점을 0으로 지정합니다
	//이 연산 방식은 암호학의 기초로 많이 활용됩니다
	//1011
	//1110
	//0101 --> 4 + 1 =5
	printf("a ^ b = %d\n", c);


	c = 6;
	printf("c = %d\n", ~c);
	//NOT 연산
	//비트를 반대로 처리함


	a = 1;
	c = a << 2;
	printf("c = %d\n", c);


	a = 3;
	c = a << 2;
	printf("c = %d\n", c);


	//shift연산
	// a << 2 일 경우 a를 비트 2칸 이동합니다. 비트 2칸은 2의 2제곱(4)을 의미합니다.
	// a * 4로 해석이 가능합니다.


	a = 16;
	c = a >> 4;
	printf("c = %d\n", c);
	// a >> 4 일 경우 a를 비트 4칸 이동합니다. 비트 4칸은 2의 4제곱(16)을 의미합니다
	// a / 16 과 동일한 결과



	

	//11 & 14는 11과 14가 가지고 있는 비트 중에서 겹치는 부분만 1로 처리, 나머지는 다 0으로
	//1011
	//1110
	//1010 --> 8 + 2 = 10


return 0;
}
```
